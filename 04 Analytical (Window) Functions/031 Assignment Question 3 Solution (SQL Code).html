SELECT

ORDER_ID,

TO_CHAR(ORDER_DATETIME,'MM-YY') as MONTH_YEAR,

ORDER_TOTAL,

AVG(ORDER_TOTAL) OVER(PARTITION BY TO_NUMBER(TO_CHAR(ORDER_DATETIME,'YYMM')) ORDER BY TO_NUMBER(TO_CHAR(ORDER_DATETIME,'YYMM')) RANGE 2 PRECEDING) AS AVG3MONTH,

ORDER_TOTAL - AVG(ORDER_TOTAL) OVER(PARTITION BY TO_NUMBER(TO_CHAR(ORDER_DATETIME,'YYMM')) ORDER BY TO_NUMBER(TO_CHAR(ORDER_DATETIME,'YYMM')) RANGE 2 PRECEDING) AS VAR

FROM

V_ORDERS;
